CREATE TABLE `hacknet` (
	`id` INT(5) NOT NULL AUTO_INCREMENT,
	`titulo` VARCHAR(64) NOT NULL COLLATE 'utf8_general_ci',
	`descripción` VARCHAR(256) NOT NULL COLLATE 'utf8_general_ci',
	`link` VARCHAR(255) NOT NULL COLLATE 'utf8_general_ci',
	PRIMARY KEY (`id`) USING BTREE
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;
CREATE TABLE `proyecto` (
	`id` INT(5) NOT NULL AUTO_INCREMENT,
	`titulo` VARCHAR(64) NOT NULL COLLATE 'utf8_general_ci',
	`descripcion` VARCHAR(256) NOT NULL COLLATE 'utf8_general_ci',
	`lenguajes` VARCHAR(100) NOT NULL COLLATE 'utf8_general_ci',
	`capacidad` INT(4) NOT NULL,
	`fecha_inicio` TIMESTAMP NOT NULL DEFAULT current_timestamp(),
	`fecha_fin` DATETIME NULL DEFAULT NULL,
	`imagen` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	PRIMARY KEY (`id`) USING BTREE
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;
CREATE TABLE `ranked` (
	`id` INT(5) NOT NULL AUTO_INCREMENT,
	`titulo` VARCHAR(64) NOT NULL COLLATE 'utf8_general_ci',
	`descripción` VARCHAR(256) NOT NULL COLLATE 'utf8_general_ci',
	`link` VARCHAR(255) NOT NULL COLLATE 'utf8_general_ci',
	PRIMARY KEY (`id`) USING BTREE
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;
CREATE TABLE `talleres` (
	`id` INT(5) NOT NULL AUTO_INCREMENT,
	`titulo` VARCHAR(64) NOT NULL COLLATE 'utf8_general_ci',
	`descripción` VARCHAR(256) NOT NULL COLLATE 'utf8_general_ci',
	`Fecha-publicacion` TIMESTAMP NOT NULL DEFAULT current_timestamp(),
	`fecha-realizacion` DATETIME NOT NULL,
	`seccion` INT(1) NOT NULL,
	`capacidad` INT(4) NOT NULL,
	`Aula` VARCHAR(16) NULL DEFAULT NULL COLLATE 'utf8_general_ci',
	PRIMARY KEY (`id`) USING BTREE
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;
CREATE TABLE `usuarios` (
	`id` INT(9) NOT NULL AUTO_INCREMENT,
	`nombre` VARCHAR(25) NOT NULL COLLATE 'utf8_general_ci',
	`apellido1` VARCHAR(25) NOT NULL COLLATE 'utf8_general_ci',
	`apellido2` VARCHAR(25) NOT NULL COLLATE 'utf8_general_ci',
	`nexp` INT(8) NOT NULL,
	`email` VARCHAR(99) NOT NULL COLLATE 'utf8_general_ci',
	`pass` VARCHAR(64) NOT NULL COLLATE 'utf8_general_ci',
	`rol` INT(1) NOT NULL,
	PRIMARY KEY (`id`) USING BTREE,
	UNIQUE INDEX `unique` (`nexp`, `email`) USING BTREE
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=6
;

CREATE TABLE `inscripcion` (
  `id` int(9) NOT NULL AUTO_INCREMENT,
  `nombre` varchar(128) NOT NULL,
  `nexp` int(8) NOT NULL,
  `email` varchar(128) NOT NULL,
  `razon` varchar(1024) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `unique` (`nexp`,`email`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
